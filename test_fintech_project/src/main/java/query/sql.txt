CREATE table "FINTECH_USER" (
    "USERNAME"         VARCHAR2(50) NOT NULL,
    "USEREMAIL"        VARCHAR2(100) NOT NULL,
    "USERPASSWORD"     VARCHAR2(20) NOT NULL,
    "USERACCESSTOKEN"  VARCHAR2(2000) NOT NULL,
    "USERREFRESHTOKEN" VARCHAR2(2000) NOT NULL,
    "USERSEQNO"        NUMBER NOT NULL,
    constraint  "FINTECH_USER_PK" primary key ("USERNAME")
)
